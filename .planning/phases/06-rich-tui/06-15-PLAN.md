   ```typescript
   import React from 'react';
   import { Text, Box, Newline } from 'ink';
   
   interface MarkdownProps {
     content: string;
   }
   
   export function Markdown({ content }: MarkdownProps) {
     const lines = content.split('\n');
     
     return (
       <Box flexDirection="column">
         {lines.map((line, i) => (
           <MarkdownLine key={i} line={line} />
         ))}
       </Box>
     );
   }
   
   function MarkdownLine({ line }: { line: string }) {
     // Headers
     if (line.startsWith('# ')) {
       return <Text bold underline>{line.slice(2)}</Text>;
     }
     if (line.startsWith('## ')) {
       return <Text bold>{line.slice(3)}</Text>;
     }
     if (line.startsWith('### ')) {
       return <Text bold dimColor>{line.slice(4)}</Text>;
     }
     
     // Code blocks (simplified)
     if (line.startsWith('```')) {
       return <Text color="cyan">{line}</Text>;
     }
     if (line.startsWith('    ') || line.startsWith('\t')) {
       return <Text color="cyan">{line}</Text>;
     }
     
     // Inline formatting
     let text = line;
     
     // Bold **text**
     text = text.replace(/\*\*(.*?)\*\*/g, '\x1b[1m$1\x1b[0m');
     
     // Italic *text*
     text = text.replace(/\*(.*?)\*/g, '\x1b[3m$1\x1b[0m');
     
     // Code `text`
     text = text.replace(/`(.*?)`/g, '\x1b[36m$1\x1b[0m');
     
     // Links [text](url) -> text (url)
     text = text.replace(/\[(.*?)\]\((.*?)\)/g, '$1 ($2)');
     
     // Wikilinks [[Page]] -> Page
     text = text.replace(/\[\[(.*?)\]\]/g, '\x1b[4m$1\x1b[0m');
     
     return <Text>{text}</Text>;
   }
   
   export function renderMarkdown(content: string): string {
     // Simple string version for non-React contexts
     return content
       .replace(/^# (.*$)/gm, '\x1b[1m\x1b[4m$1\x1b[0m')
       .replace(/^## (.*$)/gm, '\x1b[1m$1\x1b[0m')
       .replace(/^### (.*$)/gm, '\x1b[1m\x1b[2m$1\x1b[0m')
       .replace(/\*\*(.*?)\*\*/g, '\x1b[1m$1\x1b[0m')
       .replace(/\*(.*?)\*/g, '\x1b[3m$1\x1b[0m')
       .replace(/`(.*?)`/g, '\x1b[36m$1\x1b[0m')
       .replace(/\[\[(.*?)\]\]/g, '\x1b[4m$1\x1b[0m');
   }
   ```

   import React, { useEffect, useState } from 'react';
   import { Box, Text, useInput } from 'ink';
   import type { TimelineItem } from '@cortex/core/types';
   import { Markdown } from '../utils/markdown.js';
   import { open } from 'open'; // or similar
   
   interface ItemDetailProps {
     item: TimelineItem;
   }
   
   export function ItemDetail({ item }: ItemDetailProps) {
     const [scrollOffset, setScrollOffset] = useState(0);
     
     useInput((input, key) => {
       if (key.upArrow) {
         setScrollOffset(o => Math.max(0, o - 1));
       }
       if (key.downArrow) {
         setScrollOffset(o => o + 1);
       }
       if (input === 'o' && item.url) {
         // Open URL in browser
         open(item.url);
       }
     });
     
     const content = getItemContent(item);
     const contentLines = content.split('\n');
     const visibleLines = contentLines.slice(scrollOffset, scrollOffset + 20);
     
     return (
       <Box flexDirection="column" width="100%" height="100%" padding={1}>
         {/* Header */}
         <Box marginBottom={1}>
           <Text bold wrap="end">
             {item.title || 'Untitled'}
           </Text>
         </Box>
         
         <Box marginBottom={1}>
           <Text dimColor>
             {item.source} • {new Date(item.timestamp).toLocaleString()}
           </Text>
         </Box>
         
         {item.url && (
           <Box marginBottom={1}>
             <Text color="blue" underline>
               {item.url} (press 'o' to open)
             </Text>
           </Box>
         )}
         
         <Box borderStyle="single" padding={1} flexGrow={1} flexDirection="column">
           {item.source === 'obsidian' ? (
             <Markdown content={visibleLines.join('\n')} />
           ) : (
             visibleLines.map((line, i) => (
               <Text key={i}>{line}</Text>
             ))
           )}
         </Box>
         
         <Box marginTop={1}>
           <Text dimColor>
             ↑/↓ Scroll • o Open URL • Enter Back
           </Text>
         </Box>
       </Box>
     );
   }
   
   function getItemContent(item: TimelineItem): string {
     if (item.content) {
       return item.content;
     }
     
     // Try to extract from raw_data
     try {
       const raw = JSON.parse(item.raw_data || '{}');
       return raw.text || raw.description || JSON.stringify(raw, null, 2);
     } catch {
       return item.raw_data || 'No content available';
     }
   }
   ```

   ```typescript
   export default function App({ db, config }: AppProps) {
     const { exit } = useApp();
     const [selectedIndex, setSelectedIndex] = useState(0);
     const [viewMode, setViewMode] = useState<'list' | 'detail'>('list');
     
     const { items, loading, error, refresh, hasMore, loadMore } = useTimeline({
       db,
       limit: 20,
     });
     
     const selectedItem = items[selectedIndex];
     
     useInput((input, key) => {
       if (input === 'q') {
         exit();
       }
       
       if (viewMode === 'list') {
         if (key.upArrow) {
           setSelectedIndex(i => Math.max(0, i - 1));
         }
         if (key.downArrow) {
           setSelectedIndex(i => {
             const newIndex = Math.min(items.length - 1, i + 1);
             if (newIndex >= items.length - 5 && hasMore) {
               loadMore();
             }
             return newIndex;
           });
         }
         if (key.return && selectedItem) {
           setViewMode('detail');
         }
         if (input === 'r') {
           refresh();
         }
       } else {
         // Detail mode
         if (key.escape) {
           setViewMode('list');
         }
       }
     });
     
     if (loading && items.length === 0) {
       return (
         <Box>
           <Text>Loading timeline...</Text>
         </Box>
       );
     }
     
     if (viewMode === 'detail' && selectedItem) {
       return <ItemDetail item={selectedItem} />;
     }
     
     return (
       <Box flexDirection="row" height="100%">
         <TimelineList
           items={items}
           selectedIndex={selectedIndex}
           onSelect={setSelectedIndex}
         />
         
         <Box width="50%" borderStyle="single" padding={1}>
           <Text dimColor>
             {items.length} items loaded{'\n\n'}
             Controls:{'\n'}
             ↑/↓ Navigate{'\n'}
             Enter View item{'\n'}
             r Refresh{'\n'}
             q Quit
           </Text>
         </Box>
       </Box>
     );
   }
   ```

3. Behavior:

   - Enter: view item detail (full screen)
   - In detail: Escape to go back
   - Detail shows full content
