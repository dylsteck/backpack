1. Create apps/cli/src/tui/hooks/useTimeline.ts:
   ```typescript
   import { useState, useEffect, useCallback } from 'react';
   import type { Database } from '@cortex/core/db';
   import type { TimelineItem } from '@cortex/core/types';
   
   interface UseTimelineOptions {
     db: Database;
     limit?: number;
     source?: string;
   }
   
   interface UseTimelineReturn {
     items: TimelineItem[];
     loading: boolean;
     error: Error | null;
     refresh: () => void;
     hasMore: boolean;
     loadMore: () => void;
   }
   
   export function useTimeline(options: UseTimelineOptions): UseTimelineReturn {
     const { db, limit = 50, source } = options;
     
     const [items, setItems] = useState<TimelineItem[]>([]);
     const [loading, setLoading] = useState(true);
     const [error, setError] = useState<Error | null>(null);
     const [offset, setOffset] = useState(0);
     const [hasMore, setHasMore] = useState(true);
     
     const fetchItems = useCallback(async (currentOffset: number, append: boolean = false) => {
       try {
         setLoading(true);
         
         let query = db.query.timelineItems.findMany({
           orderBy: (items, { desc }) => [desc(items.timestamp)],
           limit,
           offset: currentOffset,
         });
         
         if (source) {
           query = query.where(eq(timelineItems.source, source));
         }
         
         const newItems = await query;
         
         if (append) {
           setItems(prev => [...prev, ...newItems]);
         } else {
           setItems(newItems);
         }
         
         setHasMore(newItems.length === limit);
         setError(null);
       } catch (err) {
         setError(err instanceof Error ? err : new Error(String(err)));
       } finally {
         setLoading(false);
       }
     }, [db, limit, source]);
     
     const refresh = useCallback(() => {
       setOffset(0);
       fetchItems(0, false);
     }, [fetchItems]);
     
     const loadMore = useCallback(() => {
       const newOffset = offset + limit;
       setOffset(newOffset);
       fetchItems(newOffset, true);
     }, [offset, limit, fetchItems]);
     
     useEffect(() => {
       fetchItems(0, false);
     }, [fetchItems]);
     
     return {
       items,
       loading,
       error,
       refresh,
       hasMore,
       loadMore,
     };
   }
   ```

   ```typescript
   import React from 'react';
   import { Box, Text } from 'ink';
   import type { TimelineItem } from '@cortex/core/types';
   
   interface TimelineListProps {
     items: TimelineItem[];
     selectedIndex: number;
     onSelect: (index: number) => void;
   }
   
   export function TimelineList({ items, selectedIndex, onSelect }: TimelineListProps) {
     return (
       <Box flexDirection="column" width="50%">
         <Box marginBottom={1}>
           <Text bold underline>Timeline</Text>
         </Box>
         
         {items.map((item, index) => (
           <TimelineRow
             key={item.id}
             item={item}
             isSelected={index === selectedIndex}
           />
         ))}
       </Box>
     );
   }
   
   function TimelineRow({ item, isSelected }: { item: TimelineItem; isSelected: boolean }) {
     const icon = getSourceIcon(item.source);
     const date = new Date(item.timestamp).toLocaleDateString();
     const title = item.title || 'Untitled';
     
     return (
       <Box>
         <Text backgroundColor={isSelected ? 'blue' : undefined}>
           {isSelected ? '>' : ' '} {icon} {truncate(title, 40)} <Text dimColor>{date}</Text>
         </Text>
       </Box>
     );
   }
   
   function getSourceIcon(source: string): string {
     const icons: Record<string, string> = {
       obsidian: '📄',
       farcaster: '💬',
       teller: '💰',
       chrome: '🌐',
     };
     return icons[source] || '📄';
   }
   
   function truncate(text: string, maxLength: number): string {
     if (text.length <= maxLength) return text;
     return text.slice(0, maxLength - 3) + '...';
   }
   ```

2. Create apps/cli/src/tui/components/App.tsx:
   ```typescript
   import React, { useState, useEffect } from 'react';
   import { Box, useInput, useApp } from 'ink';
   import type { Database } from '@cortex/core/db';
   import type { Config } from '@cortex/core/config';
   import { useTimeline } from '../hooks/useTimeline.js';
   import { TimelineList } from './TimelineList.js';
   
   interface AppProps {
     db: Database;
     config: Config;
   }
   
   export default function App({ db, config }: AppProps) {
     const { exit } = useApp();
     const [selectedIndex, setSelectedIndex] = useState(0);
     
     const { items, loading, error, refresh, hasMore, loadMore } = useTimeline({
       db,
       limit: 20,
     });
     
     // Keyboard handling
     useInput((input, key) => {
       if (input === 'q' || key.escape) {
         exit();
       }
       
       if (key.upArrow) {
         setSelectedIndex(i => Math.max(0, i - 1));
       }
       
       if (key.downArrow) {
         setSelectedIndex(i => {
           const newIndex = Math.min(items.length - 1, i + 1);
           // Load more when near end
           if (newIndex >= items.length - 5 && hasMore) {
             loadMore();
           }
           return newIndex;
         });
       }
       
       if (input === 'r') {
         refresh();
       }
     });
     
     if (loading && items.length === 0) {
       return (
         <Box>
           <Text>Loading timeline...</Text>
         </Box>
       );
     }
     
     if (error) {
       return (
         <Box>
           <Text color="red">Error: {error.message}</Text>
         </Box>
       );
     }
     
     return (
       <Box flexDirection="row" height="100%">
         <TimelineList
           items={items}
           selectedIndex={selectedIndex}
           onSelect={setSelectedIndex}
         />
         
         <Box width="50%" borderStyle="single" padding={1}>
           <Text dimColor>
             Select an item to view details
             {'\n\n'}
             Controls:{'\n'}
             ↑/↓ Navigate{'\n'}
             r Refresh{'\n'}
             q Quit
           </Text>
         </Box>
       </Box>
     );
   }
   ```
