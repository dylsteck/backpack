<task type="auto">
  <name>Create tRPC Router</name>
  <files>apps/server/src/trpc/router.ts</files>
  <action>
Create tRPC router with procedures:

1. Create apps/server/src/trpc/router.ts:
   ```typescript
   import { initTRPC } from '@trpc/server';
   import { z } from 'zod';
   import type { Context } from './context.js';
   import { search } from '@backpack/core/search';
   import { syncManager } from '@backpack/core/sync';
   
   const t = initTRPC.context<Context>().create();
   
   export const router = t.router;
   export const publicProcedure = t.procedure;
   
   export const appRouter = router({
     // Timeline procedures
     timeline: router({
       list: publicProcedure
         .input(z.object({
           limit: z.number().default(50),
           offset: z.number().default(0),
           source: z.enum(['obsidian', 'farcaster', 'teller', 'chrome']).optional(),
         }))
         .query(async ({ input, ctx }) => {
           let query = ctx.db.query.timelineItems.findMany({
             orderBy: (items, { desc }) => [desc(items.timestamp)],
             limit: input.limit,
             offset: input.offset,
           });
           
           if (input.source) {
             query = query.where(eq(timelineItems.source, input.source));
           }
           
           return await query;
         }),
         
       get: publicProcedure
         .input(z.object({ id: z.string() }))
         .query(async ({ input, ctx }) => {
           return await ctx.db.query.timelineItems.findFirst({
             where: eq(timelineItems.id, input.id),
           });
         }),
     }),
     
     // Items procedures
     items: router({
       list: publicProcedure
         .input(z.object({
           source: z.enum(['obsidian', 'farcaster', 'teller', 'chrome']),
           limit: z.number().default(25),
           type: z.string().optional(),
         }))
         .query(async ({ input, ctx }) => {
           return await ctx.db.query.timelineItems.findMany({
             where: and(
               eq(timelineItems.source, input.source),
               input.type ? eq(timelineItems.type, input.type) : undefined
             ),
             orderBy: (items, { desc }) => [desc(items.timestamp)],
             limit: input.limit,
           });
         }),
     }),
     
     // Search procedures
     search: router({
       query: publicProcedure
         .input(z.object({
           query: z.string(),
           limit: z.number().default(20),
           sources: z.array(z.enum(['obsidian', 'farcaster', 'teller', 'chrome'])).optional(),
         }))
         .query(async ({ input }) => {
           return await search({
             query: input.query,
             limit: input.limit,
             filters: {
               sources: input.sources,
             },
           });
         }),
     }),
     
     // Sync procedures
     sync: router({
       trigger: publicProcedure
         .input(z.object({
           sources: z.array(z.enum(['obsidian', 'farcaster', 'teller', 'chrome'])).optional(),
         }))
         .mutation(async ({ input, ctx }) => {
           initSyncers(ctx.db, ctx.config);
           
           const result = await syncManager.syncAll({
             sources: input.sources,
           });
           
           return result;
         }),
         
       status: publicProcedure
         .query(async ({ ctx }) => {
           // Return sync status for all sources
           const sources = await ctx.db.query.sources.findMany();
           return sources;
         }),
     }),
   });
   
   export type AppRouter = typeof appRouter;
   ```

2. Routes exposed:

   - items.list - List items by source
   - search.query - Semantic search
   - sync.trigger - Run sync
   - sync.status - Get sync status

3. All use @backpack/core functions

   ```typescript
   import { Elysia } from 'elysia';
   import { trpc } from '@elysiajs/trpc';
   import { appRouter } from './trpc/router.js';
   import { createContext } from './trpc/context.js';
   
   const app = new Elysia();
   
   // Health check
   app.get('/health', () => ({ status: 'ok' }));
   
   // tRPC endpoint
   app.use(
     trpc(appRouter, {
       endpoint: '/trpc',
       createContext,
     })
   );
   
   // CORS for browser access
   app.use(cors());
   
   export function startServer(port = 3000) {
     return app.listen(port, () => {
       console.log(`🚀 Server running at http://localhost:${port}`);
       console.log(`📡 tRPC endpoint: http://localhost:${port}/trpc`);
     });
   }
   
   // Start if run directly
   if (import.meta.main) {
     const port = parseInt(process.env.PORT || '3000');
     startServer(port);
   }
   
   export type App = typeof app;
   ```

2. Features:

   - Health check at /health
   - CORS enabled
   - Configurable port
   - Can be imported or run standalone
