   ```typescript
   import { createServer } from 'http';
   import { randomBytes } from 'crypto';
   import open from 'open';
   
   interface OAuthServerOptions {
     port?: number;
     provider: 'teller' | 'farcaster';
   }
   
   interface OAuthResult {
     code: string;
     state: string;
   }
   
   export async function startOAuthServer(options: OAuthServerOptions): Promise<{
     url: string;
     waitForCallback: () => Promise<OAuthResult>;
   }> {
     const port = options.port || 0; // Random available port
     const state = randomBytes(16).toString('hex');
     
     return new Promise((resolve, reject) => {
       const server = createServer((req, res) => {
         const url = new URL(req.url!, `http://localhost:${port}`);
         
         // Handle callback
         if (url.pathname === '/callback') {
           const code = url.searchParams.get('code');
           const returnedState = url.searchParams.get('state');
           const error = url.searchParams.get('error');
           
           if (error) {
             res.end(`<h1>Error: ${error}</h1><p>You can close this window.</p>`);
             server.close();
             reject(new Error(`OAuth error: ${error}`));
             return;
           }
           
           if (!code) {
             res.end('<h1>Error: No code received</h1>');
             server.close();
             reject(new Error('No authorization code received'));
             return;
           }
           
           if (returnedState !== state) {
             res.end('<h1>Error: Invalid state</h1>');
             server.close();
             reject(new Error('Invalid state parameter'));
             return;
           }
           
           res.end('<h1>Success!</h1><p>You can close this window and return to the CLI.</p>');
           server.close();
           resolve({
             waitForCallback: () => Promise.resolve({ code, state }),
           } as any);
         } else {
           res.statusCode = 404;
           res.end('Not found');
         }
       });
       
       server.listen(port, '127.0.0.1', () => {
         const actualPort = (server.address() as any).port;
         const callbackUrl = `http://127.0.0.1:${actualPort}/callback`;
         
         resolve({
           url: callbackUrl,
           waitForCallback: () => {
             return new Promise((resolve, reject) => {
               server.on('close', () => {
                 // Result will be resolved in request handler
               });
             });
           },
         });
       });
       
       server.on('error', reject);
     });
   }
   ```

2. Simplified version - better approach:
   ```typescript
   export async function startOAuthFlow(params: {
     authUrl: string;
     provider: string;
   }): Promise<{ code: string }> {
     const port = await getAvailablePort();
     const state = randomBytes(16).toString('hex');
     const callbackUrl = `http://127.0.0.1:${port}/callback`;
     
     // Build auth URL with state and redirect
     const url = new URL(params.authUrl);
     url.searchParams.set('state', state);
     url.searchParams.set('redirect_uri', callbackUrl);
     
     // Open browser
     await open(url.toString());
     
     // Wait for callback
     return waitForCallback(port, state);
   }
   
   function waitForCallback(port: number, expectedState: string): Promise<{ code: string }> {
     return new Promise((resolve, reject) => {
       const server = createServer((req, res) => {
         const url = new URL(req.url!, `http://localhost:${port}`);
         
         if (url.pathname === '/callback') {
           const code = url.searchParams.get('code');
           const returnedState = url.searchParams.get('state');
           
           if (returnedState !== expectedState) {
             res.end('Error: Invalid state');
             reject(new Error('Invalid state'));
             return;
           }
           
           res.end('<h1>Success! Close this window.</h1>');
           server.close();
           resolve({ code: code! });
         }
       });
       
       server.listen(port, '127.0.0.1');
       
       // Timeout after 5 minutes
       setTimeout(() => {
         server.close();
         reject(new Error('OAuth timeout'));
       }, 5 * 60 * 1000);
     });
   }
   ```

   ```typescript
     const applicationId = process.env.TELLER_APPLICATION_ID;
     if (!applicationId) {
       throw new Error('TELLER_APPLICATION_ID not set');
     }
     
     const baseUrl = environment === 'sandbox'
       ? 'https://auth.sandbox.teller.io'
       : 'https://auth.teller.io';
     
     console.log('Opening Teller authentication...');
     console.log(`Environment: ${environment}`);
     
     const { code } = await startOAuthFlow({
       authUrl: `${baseUrl}/authorize`,
       provider: 'teller',
     });
     
     // Exchange code for access token
     console.log('Getting access token...');
     const token = await exchangeCodeForToken(code, environment);
     
     // Store in keychain
     await setSecret('teller-access-token', token);
     await setSecret('teller-environment', environment);
     
     console.log('✓ Teller authentication successful');
     console.log('Token stored in keychain');
   }
   
   async function exchangeCodeForToken(code: string, environment: string): Promise<string> {
     // Teller uses the code directly as the access token in their OAuth implementation
     // Or you may need to make a POST request depending on Teller's specific flow
     // This is simplified - check Teller docs for exact flow
     return code;
   ```

1. Create apps/server/src/oauth/farcaster.ts:
   ```typescript
   import { startOAuthFlow } from './index.js';
   import { setSecret } from '@backpack/core/auth/keychain';
   
   export async function authenticateFarcaster() {
     // Farcaster uses Neynar for managed signers
     // Or direct signer creation
     // This example shows Neynar flow
     
     const neynarApiKey = process.env.NEYNAR_API_KEY;
     if (!neynarApiKey) {
       throw new Error('NEYNAR_API_KEY not set');
     }
     
     console.log('Opening Farcaster authentication via Neynar...');
     
     // Generate signer UUID
     const { signerUuid, deeplinkUrl } = await createNeynarSigner(neynarApiKey);
     
     // Open Warpcast app or browser
     console.log('Please approve the signer in Warpcast...');
     await open(deeplinkUrl);
     
     // Wait for approval (poll Neynar API)
     const { signer, fid } = await waitForSignerApproval(signerUuid, neynarApiKey);
     
     // Store in keychain
     await setSecret('farcaster-signer', signer);
     await setSecret('farcaster-fid', String(fid));
     
     console.log('✓ Farcaster authentication successful');
     console.log(`FID: ${fid}`);
   }
   
   async function createNeynarSigner(apiKey: string) {
     const res = await fetch('https://api.neynar.com/v2/farcaster/signer', {
       method: 'POST',
       headers: {
         'api-key': apiKey,
       },
     });
     
     const data = await res.json();
     return {
       signerUuid: data.signer_uuid,
       deeplinkUrl: data.deep_link_url,
     };
   }
   
   async function waitForSignerApproval(signerUuid: string, apiKey: string) {
     // Poll for approval
     for (let i = 0; i < 60; i++) {
       const res = await fetch(`https://api.neynar.com/v2/farcaster/signer/${signerUuid}`, {
         headers: { 'api-key': apiKey },
       });
       
       const data = await res.json();
       
       if (data.status === 'approved') {
         return {
           signer: data.signer,
           fid: data.fid,
         };
       }
       
       await new Promise(r => setTimeout(r, 2000));
     }
     
     throw new Error('Signer approval timeout');
   }
   ```

   ```typescript
   import { Command, Flags } from '@oclif/core';
   import { authenticateTeller } from '@backpack/server/oauth/teller.js';
   import { authenticateFarcaster } from '@backpack/server/oauth/farcaster.js';
   import { deleteSecret } from '@backpack/core/auth/keychain.js';
   
   export default class Auth extends Command {
     static description = 'Authenticate with external services';
     
     static args = {
       provider: Flags.string({
         required: true,
         options: ['teller', 'farcaster'],
         description: 'Service to authenticate with',
       }),
     };
     
     static flags = {
       'teller-environment': Flags.string({
         options: ['sandbox', 'production'],
         default: 'sandbox',
       }),
       logout: Flags.boolean({
         description: 'Remove authentication',
       }),
     };
     
     async run() {
       const { args, flags } = await this.parse(Auth);
       
       if (flags.logout) {
         await this.logout(args.provider);
         return;
       }
       
       switch (args.provider) {
         case 'teller':
           await authenticateTeller(flags['teller-environment']);
           break;
         case 'farcaster':
           await authenticateFarcaster();
           break;
       }
     }
     
     async logout(provider: string) {
       switch (provider) {
         case 'teller':
           await deleteSecret('teller-access-token');
           await deleteSecret('teller-environment');
           break;
         case 'farcaster':
           await deleteSecret('farcaster-signer');
           await deleteSecret('farcaster-fid');
           break;
       }
       
       this.log(`Logged out from ${provider}`);
     }
   }
   ```
