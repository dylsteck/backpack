   export default class Sync extends Command {
     static description = 'Sync data from all sources';
     
     static flags = {
       source: Flags.string({ char: 's', options: ['obsidian', 'farcaster', 'teller', 'chrome', 'all'], default: 'all' }),
       daemon: Flags.boolean({ description: 'Run as daemon (continuous sync)' }),
       interval: Flags.integer({ description: 'Sync interval in seconds (daemon mode)', default: 300 }),
       once: Flags.boolean({ description: 'Run sync once (even with sources configured)', default: true }),
       json: Flags.boolean({ description: 'Output as JSON' }),
     };
     
     async run() {
       const { flags } = await this.parse(Sync);
       
       // Initialize syncers
       const config = await getConfig();
       const db = getDatabase();
       initSyncers(db, config);
       
       if (flags.daemon) {
         await this.runDaemon(flags);
       } else {
         await this.runOnce(flags);
       }
     }
     
     async runOnce(flags) {
       const sources = flags.source === 'all' ? undefined : [flags.source];
       
       this.log(flags.json ? '' : 'Starting sync...\n');
       
       const startTime = Date.now();
       
       const result = await syncManager.syncAll({
         sources,
         onProgress: (progress) => {
           if (!flags.json) {
             this.log(`${progress.source}: ${progress.status} (${progress.itemsAdded} added)`);
           }
         },
       });
       
       const duration = Date.now() - startTime;
       
       if (flags.json) {
         this.log(JSON.stringify({ result, durationMs: duration }, null, 2));
       } else {
         this.log(`\nSync complete in ${duration}ms`);
         this.log(`Overall: ${result.overall}`);
         Object.entries(result.sourceResults).forEach(([source, progress]) => {
           this.log(`  ${source}: ${progress.itemsAdded} added, ${progress.itemsUpdated} updated`);
         });
       }
     }
     
     async runDaemon(flags) {
       this.log(`Starting daemon (sync every ${flags.interval}s)...`);
       this.log('Press Ctrl+C to stop\n');
       
       while (true) {
         await this.runOnce({ ...flags, json: false });
         this.log(`\nNext sync in ${flags.interval}s...\n`);
         await sleep(flags.interval * 1000);
       }
     }
   }
   
   function sleep(ms) {
     return new Promise(resolve => setTimeout(resolve, ms));
   }
   ```

   export default class Search extends Command {
     static description = 'Search timeline items';
     
     static args = {
       query: Args.string({ required: true, description: 'Search query' }),
     };
     
     static flags = {
       limit: Flags.integer({ char: 'n', default: 20 }),
       source: Flags.string({ char: 's', options: ['obsidian', 'farcaster', 'teller', 'chrome'], multiple: true }),
       since: Flags.string({ description: 'Filter by start date' }),
       until: Flags.string({ description: 'Filter by end date' }),
       semantic: Flags.boolean({ description: 'Use semantic search only', default: true }),
       text: Flags.boolean({ description: 'Use full-text search only', allowNo: true }),
       json: Flags.boolean({ description: 'Output as JSON' }),
     };
     
     async run() {
       const { args, flags } = await this.parse(Search);
       
       this.log(flags.json ? '' : `Searching: "${args.query}"\n`);
       
       const startTime = Date.now();
       
       const results = await search({
         query: args.query,
         filters: {
           sources: flags.source,
           startDate: flags.since ? parseDate(flags.since) : undefined,
           endDate: flags.until ? parseDate(flags.until) : undefined,
         },
         limit: flags.limit,
         useSemantic: flags.semantic && !flags.text,
         useFullText: flags.text || !flags.semantic,
       });
       
       const duration = Date.now() - startTime;
       
       if (flags.json) {
         this.log(JSON.stringify({ ...results, durationMs: duration }, null, 2));
       } else {
         if (results.results.length === 0) {
           this.log('No results found.');
           return;
         }
         
         this.log(`Found ${results.results.length} results (${duration}ms):\n`);
         
         results.results.forEach((result, i) => {
           this.log(`${i + 1}. ${formatItem(result.item)}`);
           this.log(`   Score: ${(result.score * 100).toFixed(1)}% | Match: ${result.matchType}`);
           if (result.highlights) {
             this.log(`   Highlights: ${result.highlights.join(' ... ')}`);
           }
           this.log('');
         });
       }
     }
   }
   ```
