   ```typescript
   import { spawn } from 'child_process';
   
   export async function showInPager(content: string, title?: string): Promise<void> {
     const pager = process.env.PAGER || (await findPager());
     
     return new Promise((resolve, reject) => {
       const proc = spawn(pager, [], {
         stdio: ['pipe', 'inherit', 'inherit'],
       });
       
       proc.stdin.write(content);
       proc.stdin.end();
       
       proc.on('close', resolve);
       proc.on('error', reject);
     });
   }
   
   async function findPager(): Promise<string> {
     // Check for bat, less, more in order
     const pagers = ['bat', 'less -R', 'more'];
     for (const pager of pagers) {
       try {
         await execAsync(`which ${pager.split(' ')[0]}`);
         return pager;
       } catch {}
     }
     return 'cat'; // Fallback
   }
   ```

   ```typescript
   export default class View extends Command {
     static description = 'View full details of an item';
     
     static args = {
       id: Args.string({ required: true, description: 'Item ID' }),
     };
     
     static flags = {
       json: Flags.boolean({ description: 'Output as JSON' }),
       raw: Flags.boolean({ description: 'Show raw content without formatting' }),
       'no-pager': Flags.boolean({ description: 'Disable pager' }),
     };
     
     async run() {
       const { args, flags } = await this.parse(View);
       const db = getDatabase();
       
       // Fetch item
       const item = await db.query.timelineItems.findFirst({
         where: eq(timelineItems.id, args.id),
       });
       
       if (!item) {
         this.error(`Item not found: ${args.id}`);
       }
       
       if (flags.json) {
         this.log(JSON.stringify(item, null, 2));
         return;
       }
       
       // Format content for display
       let content = this.formatItemForDisplay(item, flags.raw);
       
       // Show in pager or stdout
       if (flags['no-pager'] || !process.stdout.isTTY) {
         this.log(content);
       } else {
         await showInPager(content, item.title || 'Item Detail');
       }
     }
     
     formatItemForDisplay(item: TimelineItem, raw: boolean): string {
       if (raw) {
         return item.content || item.raw_data || 'No content';
       }
       
       let output = '';
       
       // Header
       output += `═`.repeat(80) + '\n';
       output += `${item.title || 'Untitled'}\n`;
       output += `${item.source} • ${new Date(item.timestamp).toLocaleString()}\n`;
       output += `ID: ${item.id}\n`;
       output += `═`.repeat(80) + '\n\n';
       
       // Content
       if (item.source === 'obsidian' && item.content) {
         // Render markdown (simple terminal formatting)
         output += this.renderMarkdown(item.content);
       } else {
         output += item.content || 'No content';
       }
       
       // Metadata
       output += '\n\n' + `─`.repeat(80) + '\n';
       output += `Type: ${item.type}\n`;
       output += `URL: ${item.url || 'N/A'}\n`;
       output += `External ID: ${item.external_id || 'N/A'}\n`;
       
       return output;
     }
     
     renderMarkdown(content: string): string {
       // Simple markdown to terminal formatting
       return content
         .replace(/^# (.*$)/gm, '\x1b[1m$1\x1b[0m') // Headers bold
         .replace(/^## (.*$)/gm, '\x1b[1m  $1\x1b[0m')
         .replace(/\*\*(.*?)\*\*/g, '\x1b[1m$1\x1b[0m') // Bold
         .replace(/\*(.*?)\*/g, '\x1b[3m$1\x1b[0m') // Italic
         .replace(/`(.*?)`/g, '\x1b[36m$1\x1b[0m'); // Code cyan
     }
   }
   ```

   ```typescript
   export default class Daemon extends Command {
     static description = 'Manage the Backpack daemon';
     
     static args = {
       action: Args.string({ 
         required: true, 
         options: ['start', 'stop', 'status', 'restart'],
         description: 'Action to perform',
       }),
     };
     
     static flags = {
       interval: Flags.integer({ description: 'Sync interval in seconds', default: 300 }),
     };
     
     async run() {
       const { args, flags } = await this.parse(Daemon);
       
       switch (args.action) {
         case 'start':
           await this.start(flags);
           break;
         case 'stop':
           await this.stop();
           break;
         case 'status':
           await this.status();
           break;
         case 'restart':
           await this.stop();
           await this.start(flags);
           break;
       }
     }
     
     async start(flags) {
       // Check if already running
       const pid = await this.getPid();
       if (pid && this.isProcessRunning(pid)) {
         this.log('Daemon is already running (PID: ${pid})');
         return;
       }
       
       // Start daemon process
       this.log('Starting daemon...');
       
       const child = spawn(process.execPath, [
         'bin/run',
         'sync',
         '--daemon',
         '--interval',
         String(flags.interval),
       ], {
         detached: true,
         stdio: 'ignore',
       });
       
       child.unref();
       
       // Write PID file
       await this.writePid(child.pid);
       
       this.log(`Daemon started (PID: ${child.pid})`);
       this.log(`Syncing every ${flags.interval} seconds`);
     }
     
     async stop() {
       const pid = await this.getPid();
       if (!pid) {
         this.log('Daemon is not running');
         return;
       }
       
       try {
         process.kill(pid, 'SIGTERM');
         await this.removePid();
         this.log('Daemon stopped');
       } catch (error) {
         this.error(`Failed to stop daemon: ${error.message}`);
       }
     }
     
     async status() {
       const pid = await this.getPid();
       if (pid && this.isProcessRunning(pid)) {
         this.log(`Daemon is running (PID: ${pid})`);
       } else {
         this.log('Daemon is not running');
         if (pid) {
           // Stale PID file
           await this.removePid();
         }
       }
     }
     
     getPidFile(): string {
       return join(tmpdir(), 'backpack-daemon.pid');
     }
     
     async getPid(): Promise<number | null> {
       try {
         const pid = await readFile(this.getPidFile(), 'utf8');
         return parseInt(pid.trim(), 10);
       } catch {
         return null;
       }
     }
     
     async writePid(pid: number): Promise<void> {
       await writeFile(this.getPidFile(), String(pid));
     }
     
     async removePid(): Promise<void> {
       try {
         await unlink(this.getPidFile());
       } catch {}
     }
     
     isProcessRunning(pid: number): boolean {
       try {
         process.kill(pid, 0);
         return true;
       } catch {
         return false;
       }
     }
   }
   ```
