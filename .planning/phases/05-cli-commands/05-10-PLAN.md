1. Command structure:
   ```typescript
   export default class Config extends Command {
     static description = 'Manage Backpack configuration';
     
     static flags = {
       get: Flags.string({ description: 'Get config value' }),
       set: Flags.string({ description: 'Set config value' }),
       unset: Flags.string({ description: 'Remove config value' }),
       'set-secret': Flags.string({ description: 'Set secret (stored in keychain)' }),
       json: Flags.boolean({ description: 'Output as JSON' }),
     };
     
     async run() {
       const { flags } = await this.parse(Config);
       
       if (flags.get) {
         // Get specific value
         const config = await getConfig();
         const value = flags.get.split('.').reduce((obj, key) => obj?.[key], config);
         this.log(flags.json ? JSON.stringify({ [flags.get]: value }) : value);
       } else if (flags.set) {
         // Parse key=value
         const [key, ...valueParts] = flags.set.split('=');
         const value = valueParts.join('=');
         await setConfig({ [key]: this.parseValue(value) });
         this.log(`Set ${key} = ${value}`);
       } else if (flags['set-secret']) {
         // Set secret in keychain
         const [key, value] = flags['set-secret'].split('=');
         await setSecret(key, value);
         this.log(`Set secret ${key} (stored in keychain)`);
       } else if (flags.unset) {
         // Remove config value
         await setConfig({ [flags.unset]: undefined });
         this.log(`Unset ${flags.unset}`);
       } else {
         // Show all config
         const config = await getConfig();
         if (flags.json) {
           this.log(JSON.stringify(config, null, 2));
         } else {
           this.log('Current configuration:');
           this.log(JSON.stringify(config, null, 2));
         }
       }
     }
     
     parseValue(value: string): any {
       // Parse numbers, booleans, JSON
       if (value === 'true') return true;
       if (value === 'false') return false;
       if (!isNaN(Number(value))) return Number(value);
       try { return JSON.parse(value); } catch { return value; }
     }
   }
   ```
